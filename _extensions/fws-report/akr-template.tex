\DeclareRobustCommand{\TitleNoPipesOneLine}[1]{%
  \StrSubstitute{#1}{| }{}[\tmpTitleA]%
  \StrSubstitute{\tmpTitleA}{|}{}[\tmpTitleB]%
  \StrSubstitute{\tmpTitleB}{\\}{ }[\tmpTitleC]%
  \StrSubstitute{\tmpTitleC}{\newline}{ }[\tmpTitleD]%
  \tmpTitleD%
}
