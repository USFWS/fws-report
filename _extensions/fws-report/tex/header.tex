% Section formatting: ONLY # headings (\section): all caps with light tracking.
\newcommand{\FwsSectionStyle}[1]{%
  {\RobotoBold\addfontfeatures{LetterSpace=10}\MakeUppercase{#1}}%
}

\titleformat{\section}{\bfseries\fontsize{14}{16}\selectfont}{}{0pt}{\FwsSectionStyle}
\titlespacing*{\section}{0pt}{14pt}{8pt}

\titleformat{\subsection}{\bfseries\fontsize{12}{14}\selectfont}{}{0pt}{}
\titlespacing*{\subsection}{0pt}{12pt}{6pt}

\titleformat{\subsubsection}{\itshape\fontsize{11}{13}\selectfont}{}{0pt}{}
\titlespacing*{\subsubsection}{0pt}{10pt}{4pt}

\captionsetup{
  font=small,
  labelfont=normalfont,
  textfont=normalfont,
  justification=raggedright,
  singlelinecheck=false
}

\captionsetup[figure]{position=below, skip=6pt}
